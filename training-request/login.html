'admin.html' : 'index.html'); // Don't allow regular users to access admin pages if (redirectTo.includes('admin') && !isAdmin) { window.location.href = 'index.html'; return; } window.location.href = redirectTo; return; } // Handle Discord login const discordLogin = document.getElementById('discordLogin'); discordLogin.addEventListener('click', (e) => { e.preventDefault(); initiateDiscordLogin(); }); // Check for error messages in session storage or URL const urlParams = new URLSearchParams(window.location.search); let error = urlParams.get('error'); // Check session storage for error messages first const sessionError = sessionStorage.getItem('loginError'); if (sessionError) { error = sessionError; sessionStorage.removeItem('loginError'); // Clear after reading } if (error) { showError(decodeURIComponent(error)); } }); function initiateDiscordLogin() { // Clear any existing session data sessionStorage.clear(); // Build the Discord OAuth2 URL const clientId = '1381993589663338637'; // FSRP Bot Client ID const redirectUri = encodeURIComponent(`${window.location.origin}/training-request/auth/discord/callback.html`); const scopes = encodeURIComponent('identify email'); const state = Math.random().toString(36).substring(2); // Store state for CSRF protection sessionStorage.setItem('oauth_state', state); // Build auth URL const authUrl = `https://discord.com/api/oauth2/authorize?client_id=${clientId}&redirect_uri=${redirectUri}&response_type=token&scope=${scopes}&state=${state}`; console.log('Initiating Discord OAuth:', authUrl); // Redirect to Discord window.location.href = authUrl; } function showError(message) { const errorEl = document.getElementById('errorMessage'); errorEl.textContent = message; errorEl.style.display = 'block'; // Auto-hide error after 5 seconds setTimeout(() => { errorEl.style.display = 'none'; }, 5000); }
